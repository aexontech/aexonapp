   - Implementasikan menggunakan Web Crypto API (built-in browser, tidak perlu library):
```typescript
     async function deriveKey(userId: string): Promise<CryptoKey> {
       const encoder = new TextEncoder();
       const keyMaterial = await crypto.subtle.importKey(
         "raw", encoder.encode(`aexon_${userId}`),
         "PBKDF2", false, ["deriveKey"]
       );
       return crypto.subtle.deriveKey(
         { name: "PBKDF2", salt: encoder.encode("aexon_salt_v2"), iterations: 100000, hash: "SHA-256" },
         keyMaterial,
         { name: "AES-GCM", length: 256 },
         false, ["encrypt", "decrypt"]
       );
     }
     
     export async function saveUserData(userId: string, dataKey: string, data: any): Promise<void> {
       const key = await deriveKey(userId);
       const iv = crypto.getRandomValues(new Uint8Array(12));
       const encoded = new TextEncoder().encode(JSON.stringify(data));
       const encrypted = await crypto.subtle.encrypt({ name: "AES-GCM", iv }, key, encoded);
       const stored = { iv: Array.from(iv), data: Array.from(new Uint8Array(encrypted)) };
       localStorage.setItem(`aexon_${dataKey}_${userId}`, JSON.stringify(stored));
     }
     
     export async function loadUserData<T>(userId: string, dataKey: string): Promise<T | null> {
       const raw = localStorage.getItem(`aexon_${dataKey}_${userId}`);
       if (!raw) return null;
       try {
         const { iv, data } = JSON.parse(raw);
         const key = await deriveKey(userId);
         const decrypted = await crypto.subtle.decrypt(
           { name: "AES-GCM", iv: new Uint8Array(iv) },
           key, new Uint8Array(data)
         );
         return JSON.parse(new TextDecoder().decode(decrypted));
       } catch {
         // Fallback: coba parse legacy unencrypted data
         try { return JSON.parse(raw); } catch { return null; }
       }
     }
```
